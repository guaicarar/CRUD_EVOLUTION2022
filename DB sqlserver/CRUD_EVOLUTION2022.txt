CREATE TABLE dbo.PEDIDO (
  PedID int NOT NULL,
  PedUsu int NULL,

      ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON)
)
ON [PRIMARY]
GO
